1. 打印X
   我的解法：
   ```c
   int main()
{
    int a=0;
    while (scanf("%d",&a) != EOF)
    {
        if (a % 2 == 1)
        {
            int tem1 = a - 2;
            int tem2 = 1;
            for (int i = 1; i <= a; i++)
            {
                if (i < (a + 1) / 2)
                {
                    for (int j = 1; j < i; j++)
                    {
                        printf(" ");
                    }
                    printf("*");
                    for (int j = 1; j <= tem1; j++)
                    {
                        printf(" ");
                    }
                    printf("*");
                    for (int j = 1; j < i; j++)
                    {
                        printf(" ");
                    }
                    printf("\n");
                    tem1 -= 2;
                }
                else if (i == (a + 1) / 2)
                {
                    for (int j = 1;j <= a / 2;j++)
                    {
                        printf(" ");
                    }
                    printf("*");
                    for (int j = 1;j <= a / 2;j++)
                    {
                        printf(" ");
                    }
                    printf("\n");
                }
                else if (i > (a + 1) / 2)
                {
                    for (int j = 1;j <= a - i;j++)
                    {
                        printf(" ");
                    }
                    printf("*");
                    for (int j = 1;j <= tem2;j++)
                    {
                        printf(" ");
                    }
                    printf("*");
                    for (int j = 1;j <= a - i;j++)
                    {
                        printf(" ");
                    }
                    printf("\n");
                    tem2 += 2;
                }
            }
        }
    }
}
   ```
核心思路是寻求每一行的规律，机械式的照抄需求，虽然也能解决需求，但是代码过于冗杂，而且可读性很差。实际上可以先找出相关的数学规律，大大地提高代码效率和可读性。
优化解法：
```c
int main()
{
	int a = 0;
	while (scanf("%d", &a) != EOF)
	{
		for (int i = 1;i <= a;i++)
		{
			for (int j = 1;j <= a;j++)
			{
				if (i == j || i ==a - j + 1)
				{
					printf("*");
				}
				else
				{
					printf(" ");
				}
			}
			printf("\n");
		}
	}
}
```
>[!tips]
>学编程重要的不是你学会几门语言，语言只是工具，重要的是怎么样去使用代码，重要的是思维。

2. 打印正方形
   ```c
   #define _CRT_SECURE_NO_WARNINGS
	#include <stdio.h>
	int main()
	{
	    int a;
	    while (scanf("%d", &a) != EOF)
	    {
	        for (int i = 1;i <= a;i++)
	        {
	            for (int j = 1;j <= a;j++)
	            {
	                if (i == 1 || i == a || j == 1 || j == a)
	                {
	                    printf("* ");
	                }
	                else
	                {
	                    printf("  ");
	                }
	            }
	            printf("\n");
	        }
	    }
	    return 0;
	}
   ```
解决思路大同小异，也可以尝试使用二维数组来实现：
```c
int main()
{
	int a=0;
	char arr[20][20];
	while (scanf("%d", &a) != EOF)
	{
		for (int i = 0;i < a;i++)
		{
			for (int j = 0;j < a;j++)
			{
				if (i == 0 || i == a - 1 || j == 0 || j == a - 1)
					arr[i][j] = '*';
				else
					arr[i][j] = ' ';
			}
		}
		for (int i = 0;i < a;i++)
		{
			for (int j = 0;j < a;j++)
			{
				printf("%c ", arr[i][j]);
			}
			printf("\n");
		}
	}
	return 0;
}
```
>[!bug]
>注意第一次运行的时候，没有打印出来任何结果，这是因为单引号`''`只能包含一个字符，而我刚开始写的时候是以`'  '`和`'* '`的形式，这里面有了两个字符，所以不能正常输出。
