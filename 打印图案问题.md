1. 打印X
   我的解法：
   ```c
   int main()
{
    int a=0;
    while (scanf("%d",&a) != EOF)
    {
        if (a % 2 == 1)
        {
            int tem1 = a - 2;
            int tem2 = 1;
            for (int i = 1; i <= a; i++)
            {
                if (i < (a + 1) / 2)
                {
                    for (int j = 1; j < i; j++)
                    {
                        printf(" ");
                    }
                    printf("*");
                    for (int j = 1; j <= tem1; j++)
                    {
                        printf(" ");
                    }
                    printf("*");
                    for (int j = 1; j < i; j++)
                    {
                        printf(" ");
                    }
                    printf("\n");
                    tem1 -= 2;
                }
                else if (i == (a + 1) / 2)
                {
                    for (int j = 1;j <= a / 2;j++)
                    {
                        printf(" ");
                    }
                    printf("*");
                    for (int j = 1;j <= a / 2;j++)
                    {
                        printf(" ");
                    }
                    printf("\n");
                }
                else if (i > (a + 1) / 2)
                {
                    for (int j = 1;j <= a - i;j++)
                    {
                        printf(" ");
                    }
                    printf("*");
                    for (int j = 1;j <= tem2;j++)
                    {
                        printf(" ");
                    }
                    printf("*");
                    for (int j = 1;j <= a - i;j++)
                    {
                        printf(" ");
                    }
                    printf("\n");
                    tem2 += 2;
                }
            }
        }
    }
}
   ```
核心思路是寻求每一行的规律，机械式的照抄需求，虽然也能解决需求，但是代码过于冗杂，而且可读性很差。实际上可以先找出相关的数学规律，大大地提高代码效率和可读性。
优化解法：
```c
int main()
{
	int a = 0;
	while (scanf("%d", &a) != EOF)
	{
		for (int i = 1;i <= a;i++)
		{
			for (int j = 1;j <= a;j++)
			{
				if (i == j || i ==a - j + 1)
				{
					printf("*");
				}
				else
				{
					printf(" ");
				}
			}
			printf("\n");
		}
	}
}
```
>[!tips]
>学编程重要的不是你学会几门语言，语言只是工具，重要的是怎么样去使用代码，重要的是思维。
